}

country_event = {
    id = tcharialt.5
	title = tcharialt.5.t
	desc = tcharialt.5.d
	picture = EFFELAI_BLOOM_ANIMATED_eventPicture

    is_triggered_only = yes

    trigger = {
        NOT = {
            has_country_modifier = temple_of_the_new_moon_overlord
        }
        2244 = {
            has_great_project = {
                type = temple_of_the_new_moon
                tier = 1
            }
            owner = {
                is_subject_of = ROOT
            }
        }
    }


    option = { #cool
        name = tcharialt.5.a 
        ai_chance = { factor = 100 }
        trigger = {
            always = yes
        }
        immediate_effect = {
            add_country_modifier = {
                name = "temple_of_the_new_moon_overlord"
                duration = -1
            }
        }
        add_prestige = 10
    }
   
    
}
